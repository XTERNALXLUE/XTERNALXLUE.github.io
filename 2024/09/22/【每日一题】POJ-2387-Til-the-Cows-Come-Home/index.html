int n, m, g[N][N], dis[N];
bool bk[N];

void dijkstra() {
    dis[1] = 0;
    for(int i = 1; i < n; i++) {
        int t = -1;
        for(int j = 1; j <= n; j++) 
            if(!bk[j] && (t == -1 || dis[j] < dis[t])) t = j;

        bk[t] = 1;

        for(int j = 1; j <= n; j++) {
            dis[j] = min(dis[j], dis[t] + g[t][j]);
        }
    }
}

void solve() {
    memset(g, 0x3f, sizeof g);
    memset(dis, 0x3f, sizeof dis);
    cin >> m >> n;
    for(int i = 0, a, b, c; i < m; i++) {
        cin >> a >> b >> c;
        g[a][b] = g[b][a] = min(g[a][b], c);
    }

    dijkstra();

    cout << dis[n] << endl;
}
